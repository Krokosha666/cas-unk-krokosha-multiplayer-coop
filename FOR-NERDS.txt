# Port Forwarding:
This mod uses UDP protocol.
Default port: 7790 

# Console commands:
This mod adds mutliplayer-exclusive console commands.

Write "krok help" into the console to see them all.

"krok rule sv_cheats 1" to enable all commands (server-only)
"krok rule AllowClientCheatCommands 1" to enable all commands for clients (none of them are synced, only tp works)
"krok maxplayers [number]" (default is 20)

"krok rules" to see game rules
"krok rule [rule name] [value]" is for:
	- enable permadeath
	- re-enable changing time speed 
	- re-enable sleeping 
	- and much more things for people to tweak the game into a better experience 
	- at the end of the file all rules are explained 
"krok kick [name]" to forcibly disconnect a player
also some commands that require player name as input can accept macros: 
 - "@m" (local player)  
 - "@a" (everyone)
 - "@r" (random)
 - "@c" (anyone at your cursor location (or the nearest player) )
 - so you can do "krok heal @a" - which heals everyone
 - or you can do "krok heal @c" - which heals someone at your cursor 

"krok give" command is more advanced way to spawn items. it automatically loads any guns or magazines.


# Keybinds:
"krokosha_coop_pointfingerat" default: Mouse2
"krokosha_coop_chat" default: Slash

Are added to the "keybinds.json" automatically.


# Available startup os command arguments:  
--ksmulti-immidiate-start	// start the world as soon as the game initializes 
--ksmulti-setstarttutorial  // requires immidiate start, makes the world start in tutorial
--ksmulti-verbose			// logs more info about synchronization (useful for reporing bugs)
--ksmulti-ip-port 127.0.0.1:7790   // override ip and port
--ksmulti-setname My_Custom_User_Name  // override username
--ksmulti-starthost			// start host once the game initializes (on the specified port)
--ksmulti-startclient		// start client once the game initializes (it tries to connect to the specified ip instantly)
--ksmulti-startserver		// EXPERIMENTAL // start dedicated server once the game initializes (on the specified port), also sets "AutoContinue" rule to true automatically
--ksmulti-min-plr-count	[int]	// for AutoContinue rule, it tells dedicated server what is the minimum player count to automatically start the run (does not affect run continue)
--ksmulti-setrule [rule] [value] // set a game rule before start, in in-game console enter "krok rules" to see them all

DEDICATED SERVER EXAMPLE: 
"CasualtiesUnknown.exe -screen-fullscreen 0 -screen-height 360 -screen-width 640 --ksmulti-ip-port 127.0.0.1:7790 --ksmulti-setrule AllowJoinWhileInGame 1 --ksmulti-setrule PLAYER_COUNT_LIMIT 64 --ksmulti-setname Krokosha --ksmulti-startserver --ksmulti-verbose --ksmulti-min-plr-count 4 --ksmulti-immidiate-start"

# All game rules:

bool sv_cheats = false; // allow running all console commands for the host
bool AllowClientCheatCommands = false; // allow running all console commands for all clients (none of them are synced btw)
ushort PLAYER_COUNT_LIMIT = 20;  
bool EnableNametags = true; // Shows names of players under their characters 
bool DisableNametagsIfUnchipped = true; 
bool EnableChatbox = true; // In-game chat history is shown.
bool OnlyProximityChat = false; // if enabled you can only see messages of players in your 64 tile radius.
bool OnlyProximityChatIfUnchipped = true; // if enabled unchipped players can only see messages of players in 64 tile radius.
bool UnchippedIsIndividual = true;  // in pre-run settings, everyone can choose on their own to be chipped or no. if this is "false", everyone is same as host
bool ForNextLayerAllMustReachEnd = true; // player that finishes a layer must wait for everyone else, else , host will get prompted to continue layer even if they are very far behind
bool CanAccessOthersInventory = false;  // and steal stuff from other inventory even when theyre conscious
bool DisableSleep = true; // energy is always forced 100%
bool DisableTimeManipulation = true;  // kinda broken, allows time manipulation (of course its synchronized but i find it too chaotic to let everyone spam fast forward + the game also auto winds time in specific scenarios so its completely broken)
bool SpeechImpairedTextChat = true; // applies same distortions to text as default talker from current health.  and also disable text chat if mindwiped in UPDATE1 (but can still see )
bool CommunicateWithTheDead = true; // chat is separated from dead if false (forced true in tutorial)
bool Permadeath = false; // disable all forms of non-cheat respawn
bool ReviveOnNextLevel = true; // respawn all dead players after reaching next layer
bool FriendlyFire = false;
bool AllowJoinWhileInGame = false; // potentially unstable but i think it works now 
float AdditionalHealthRegen = 1.0; // speed up regen for broken bones, dislocations and stuff like that (because the intended way to play this mod is without speeding up time) 
float AdditionalHealthDecay = 1.0; // speeds up brain damage from lack of oxygen, so that death isnt so slow  
bool AutoContinue = false; // auto-continues the layer depending on the "ForNextLayerAllMustReachEnd" rule,  and in main menu it auto-starts the game after 10 seconds when a first player joined. and auto-exits to main menu if all players disconnected
